\begin{exercise}[Integral]
Compute $\dint_0^1 2x\dx$.
\mysskip
\begin{solution}
$\dint_0^1 2x\dx = \tsol{1}$.
\end{solution}
\end{exercise}

\begin{exercise}[Vector notation]
Let $\vec{R}(t)=\langle t,t^2\rangle$. Compute $\dvRdt$.
\mysskip
\begin{solution}
$\dvRdt = \tsol{\langle 1,2t\rangle}$.
\end{solution}
\end{exercise}
