\section{A few sample statements}

\begin{defn}[Inner product space]{def:inner-product}
An \emph{inner product space} is a vector space $V$ over $\R$ or $\C$ equipped with a function
$\la \cdot,\cdot \ra : V\X V \to \R$ or $\C$ satisfying the usual axioms.
\end{defn}

\begin{axiom}[Order axiom]{ax:order}
For every $x \in \R$, exactly one of the following holds: $x<0$, $x=0$, or $x>0$.
\end{axiom}

\begin{theo}[Power rule]{thm:derivative-power}
If $f(x)=x^n$ for $n\in\N$, then $\dfrac{\D}{\D x}f(x)=nx^{n-1}$.
\end{theo}

\begin{prf}{thm:derivative-power}
One proves this using a binomial theorem identity. The displayed derivative
notation in this class can be written as $\ddx(x^n)=nx^{n-1}$.
\end{prf}

\begin{prop}[A norm identity]{prop:norm}
For vectors $\vec{u}, \vec{v}\in V$, the quantity $\tripnorm \vec{u}+\vec{v} \tripnorm$ depends continuously on $\vec{u}$ and $\vec{v}$.
\end{prop}

\begin{claim}[A simple estimate]{claim:estimate}
If $|x|\le 1$, then $|x^2|\le 1$.
\end{claim}

\begin{exmpl}[A derivative computation]{ex:derivative}
If $f(x)=x^3$, then $\dfdx=3x^2$ and $f(x)\eval_{0}^{1}=1$.
\end{exmpl}

\section{Exercises and solutions}

\begin{exercise}[Warm-up]
Compute $\dint_0^1 x^2\dx$ and compare your answer with the statement of \cref{thm:derivative-power}.

\begin{solution}
We have $\dint_0^1 x^2\dx = \left.\frac{x^3}{3}\right\vert_{0}^{1}=\tsol{\frac13}$.
\end{solution}
\end{exercise}

\begin{exercise}[Vectors]
Let $\vec{R}(t)=\langle t,t^2,t^3\rangle$. Write down $\dvRdt$, $\dxdt$, $\dydt$, and $\dzdt$.

\begin{solution}
$\dvRdt=\langle 1,2t,3t^2\rangle$, while $\dxdt=1$, $\dydt=2t$, and $\dzdt=3t^2$.
\end{solution}
\end{exercise}
